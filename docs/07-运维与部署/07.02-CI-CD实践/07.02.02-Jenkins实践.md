## 2. Jenkins Job配置

### 2.1 Pipeline Job

```groovy
// Jenkinsfile
pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Test') {
            steps {
                sh 'pytest tests/'
            }
        }

        stage('Deploy') {
            steps {
                sh 'ansible-playbook deploy.yml'
            }
        }
    }
}
```

---

## 3. Pipeline脚本

### 3.1 多环境部署

```groovy
pipeline {
    agent any

    parameters {
        choice(name: 'ENVIRONMENT', choices: ['dev', 'test', 'prod'])
    }

    stages {
        stage('Deploy') {
            steps {
                sh "ansible-playbook deploy.yml -i inventory/${params.ENVIRONMENT}.ini"
            }
        }
    }
}
```
